Bubble Sort

 

# #![allow(unused_variables)]
#fn main() {
use std::ops::BitXor;

pub fn swap<T>(array: &mut Vec<T>, from: usize, to: usize)
pub fn swap_by_xor<T>(array: &mut Vec<T>, from: usize, to: usize)
where
    T: Copy + BitXor<Output = T>,
{
    // XOR trick can be used only for numbers
    array[from] = array[from] ^ array[to];
    array[to] = array[from] ^ array[to];
    array[from] = array[from] ^ array[to];

pub fn bubble_sort<T, U>(array: &mut Vec<T>, comparator: U)
where
    T: Copy + BitXor<Output = T>,
    U: Fn(T, T) -> bool,
{
    let length = array.len();
    for i in 0..(length - 1) {
        for j in (i + 1)..length {
            if comparator(array[j], array[i]) {
                swap(array, i, j);
            }
        }
    }
}

let mut v = vec![20, 12, 45, 19, 91, 55];
#println!("before: {:?}", v);
bubble_sort(&mut v, |a, b| a < b);
#println!("sorted: {:?}", v);
#assert_eq!(vec![12, 19, 20, 45, 55, 91], v)
#}

Insertion Sort

 

# #![allow(unused_variables)]
#fn main() {
pub fn insertion_sort<T: Copy + PartialOrd, F: Fn(&T, &T) -> bool>(
    array: &mut [T],
    comparator: &F,
) {
    for i in 1..array.len() {
        let num_to_insert = array[i];
        let mut compare_index = i;
        while compare_index > 0 && comparator(&num_to_insert, &array[compare_index - 1]) {
            array[compare_index] = array[compare_index - 1];
            compare_index -= 1;
        }

let mut v = vec![20, 12, 45, 19, 91, 55];
#println!("before: {:?}", v);
insertion_sort(&mut v, |a, b| a < b);
#println!("sorted: {:?}", v);
#assert_eq!(vec![12, 19, 20, 45, 55, 91], v)
#}

Quick Sort

 

# #![allow(unused_variables)]
#fn main() {
fn partition<T: Copy + PartialOrd>(v: &mut [T]) -> usize {
    let len = v.len();
    let pivot_index = len / 2;

    v.swap(pivot_index, len - 1);

    let mut store_index = 0;
    for i in 0..len - 1 {
        if &v[i] < &v[len - 1] {
            v.swap(i, store_index);
            store_index += 1;
        }
    }

    v.swap(store_index, len - 1);
    store_index
}

pub fn quick_sort<T: Copy + PartialOrd>(v: &mut [T]) {
    let len = v.len();
    if len >= 2 {
        let pivot_index = partition(v);
        quick_sort(&mut v[0..pivot_index]);
        quick_sort(&mut v[pivot_index + 1..len]);
    }
}

let mut v = vec![20, 12, 45, 19, 91, 55];
#println!("before: {:?}", v);
quick_sort(&mut v);
#println!("sorted: {:?}", v);
#assert_eq!(vec![12, 19, 20, 45, 55, 91], v)
#}

#fn main() {
fn merge<T: Copy + PartialOrd>(x1: &[T], x2: &[T], y: &mut [T]) {
    let mut i = 0;
    let mut j = 0;
    let mut k = 0;
    while i < x1.len() && j < x2.len() {
        if x1[i] < x2[j] {
            y[k] = x1[i];
            k += 1;
            i += 1;
        } else {
            y[k] = x2[j];
            k += 1;
            j += 1;
        }
    }
    if i < x1.len() {
        y[k..].copy_from_slice(&x1[i..]);
    }
    if j < x2.len() {
        y[k..].copy_from_slice(&x2[j..]);
    }
}
